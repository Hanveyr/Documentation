-Introduction-

What are awk and sed?-

    awk: A powerful programming language designed for text processing and typically used as a data extraction and reporting tool.
    sed: A stream editor used to perform basic text transformations on an input stream (a file or input from a pipeline).




Brief History-

    awk: Developed in the 1970s at Bell Labs, named after its authors Aho, Weinberger, and Kernighan.

    sed: Developed in 1973 by Lee E. McMahon of Bell Labs and has since become a standard tool in Unix-like operating systems.









-Basic Syntax and Usage-
awk Basic Usage

    Syntax:
awk 'pattern { action }' file

Example:
    awk '{ print $1 }' file.txt  # Print the first field of each line





-sed Basic Usage-

    Syntax:
sed 'script' file

Example:
    sed 's/old/new/' file.txt  # Replace the first occurrence of "old" with "new" in each line







-awk Common Options-


    Field Separator:
  awk -F',' '{ print $1 }' file.csv  # Use comma as the field separator


Print Specific Lines:
awk 'NR==3' file.txt  # Print the third line of the file


Pattern Matching:
    awk '/pattern/ { print $1 }' file.txt  # Print the first field of lines matching the pattern






-sed Common Options-


    In-Place Editing:
sed -i 's/old/new/' file.txt  # Replace "old" with "new" in the file and save changes



Print Only Matching Lines:
sed -n '/pattern/p' file.txt  # Print only lines containing the pattern



Delete Lines:
    sed '/pattern/d' file.txt  # Delete lines containing the pattern




-Advanced Options-

awk Advanced Options

    BEGIN and END Blocks:
awk 'BEGIN { print "Start" } { print $1 } END { print "End" }' file.txt




Arithmetic Operations:
awk '{ sum += $1 } END { print sum }' file.txt  # Calculate the sum of the first field



User-Defined Functions:
    awk 'function square(x) { return x * x } { print square($1) }' file.txt




s-ed Advanced Options-

    Multiple Commands:
sed -e 's/old/new/' -e 's/foo/bar/' file.txt  # Perform multiple substitutions



Insert and Append Text:
sed '1i\New first line' file.txt  # Insert a line at the beginning of the file
sed '$a\New last line' file.txt   # Append a line at the end of the file



Using Hold Space:
    sed -e '/start/{:a;N;/end/!ba};/start/,/end/!d' file.txt  # Delete everything between "start" and "end" patterns

----Common Errors and Troubleshooting--


    Syntax Errors:

Permission Denied:

    Problem: Insufficient permissions to edit files.

#!/usr/bin/awk -f
# error_handling.awk

BEGIN {
    # Set error handling variables
    ERROR_LOG = "/var/log/awk_errors.log"
    WARN_COUNT = 0
    ERROR_COUNT = 0
}

{
    # Try to process line, catch errors
    if (length($0) > 1000) {
        printf "WARNING: Line %d too long\n", NR > ERROR_LOG
        WARN_COUNT++
        next
    }

    # Validate field count
    if (NF != EXPECTED_FIELDS) {
        printf "ERROR: Line %d has wrong field count\n", NR > ERROR_LOG
        ERROR_COUNT++
        next
    }

    # Process valid lines
    process_line()
}

END {
    # Report error statistics
    printf "Processing complete: %d warnings, %d errors\n", WARN_COUNT, ERROR_COUNT > ERROR_LOG
}

#!/usr/bin/awk -f
# log_analysis.awk

# Parse complex log format
function parse_log() {
    split($0, parts, "|")
    timestamp = parts[1]
    level = parts[2]
    message = parts[3]

    # Store in time-series format
    if (level == "ERROR") {
        errors[timestamp]++
        error_msgs[timestamp, errors[timestamp]] = message
    }
}

# Calculate error rates
function calculate_metrics() {
    for (t in errors) {
        rate = errors[t] / total_messages * 100
        printf "%s: Error Rate %.2f%%\n", t, rate
    }
}

# Main processing
{
    total_messages++
    parse_log()
}

END {
    calculate_metrics()
}

#!/bin/bash

# Test suite configuration
TEST_DIR="./tests"
LOG_FILE="test_results.log"

# Test runner function
run_test() {
    local test_name="$1"
    local input="$2"
    local expected="$3"
    local script="$4"

    echo "Running test: $test_name"
    result=$(echo "$input" | awk "$script")
    if [[ "$result" == "$expected" ]]; then
        echo "✓ PASS: $test_name" | tee -a "$LOG_FILE"
        return 0
    else
        echo "✗ FAIL: $test_name" | tee -a "$LOG_FILE"
        echo " Expected: $expected" | tee -a "$LOG_FILE"
        echo " Got: $result" | tee -a "$LOG_FILE"
        return 1
    fi
}

Debugging Framework
Awk

#!/usr/bin/awk -f
# debug_framework.awk

BEGIN {
    DEBUG_LEVELS["ERROR"] = 1
    DEBUG_LEVELS["WARN"] = 2
    DEBUG_LEVELS["INFO"] = 3
    DEBUG_LEVELS["DEBUG"] = 4

    CURRENT_DEBUG_LEVEL = DEBUG_LEVELS["INFO"]
}

function debug(message, level) {
    if (level <= CURRENT_DEBUG_LEVEL) {
        printf "[%s] %s:%d - %s\n",
        get_level_name(level),
        FILENAME,
        NR,
        message > "/dev/stderr"
    }
}

function get_level_name(level) {
    for (name in DEBUG_LEVELS) {
        if (DEBUG_LEVELS[name] == level) {
            return name
        }
    }
    return "UNKNOWN"
}

#!/usr/bin/awk -f
# Standard awk script structure

# 1. Constants and configuration
BEGIN {
    FS = ","
    OFS = "|"
    MAX_FIELD_LENGTH = 80
    ERROR_LOG = "/var/log/awk.log"
}

# 2. Function declarations
function validate_record(record, result) {
    result = 1
    for (i = 1; i <= NF; i++) {
        if (length($i) > MAX_FIELD_LENGTH) {
            result = 0
            break
        }
    }
    return result
}

function process_record() {
    if (validate_record($0)) {
        print_record()
    } else {
        log_error("Invalid record length")
    }
}

# 3. Main processing
{
    process_record()
}

# 4. Cleanup and reporting
END {
    generate_report()
}

# Benchmark function
benchmark() {
    command=$1
    iterations=$2
    total_time=0

    for ((i=1; i<=iterations; i++)); do
        start_time=$(date +%s%N)
        eval "$command" >/dev/null
        end_time=$(date +%s%N)
        total_time=$((total_time + end_time - start_time))
    done

    echo "Average time: $(((total_time/iterations)/1000000)) ms"
}

# Dijkstra's shortest path
function shortest_path(start, end, dist, prev, Q) {
    # Initialize distances
    for (v in vertices) {
        dist[v] = 999999
        prev[v] = ""
        Q[v] = 1
    }
    dist[start] = 0

    while (length(Q) > 0) {
        # Find minimum distance
        min_dist = 999999
        for (v in Q) {
            if (dist[v] < min_dist) {
                min_dist = dist[v]
                u = v
            }
        }

        delete Q[u]

        for (v in vertices) {

                if (alt < dist[v]) {
                    dist[v] = alt
                    prev[v] = u
                }
            }
        }
    }

    return dist[end]
}

    delete max_vals
    delete sum_vals
    delete sum_squares

    delete categories

# First pass: collect statistics
function collect_stats() {
    for (i = 1; i <= NF; i++) {
        if ($i ~ /^[0-9]+\.?[0-9]*$/) {
            # Numeric field
            if (!(i in min_vals) || $i < min_vals[i]) min_vals[i] = $i
            if (!(i in max_vals) || $i > max_vals[i]) max_vals[i] = $i
            sum_vals[i] += $i
            sum_squares[i] += $i * $i
            counts[i]++
        } else {
            # Categorical field
            if (ONE_HOT) categories[i][$i] = 1
        }
    }
}

# Normalize numeric value
function normalize(value, field) {
    if (max_vals[field] == min_vals[field]) return 0
    return (value - min_vals[field]) / (max_vals[field] - min_vals[field])
}

function standardize(value, field) {
    mean = sum_vals[field] / counts[field]
    variance = (sum_squares[field] - (sum_vals[field]^2) / counts[field]) / (counts[field] - 1)
    return (value - mean) / sqrt(variance)
}

NR == FNR {
    collect_stats()
    next
}

{
    output = ""

                value = normalize($i, i)
            } else if (STANDARDIZE) {
                value = standardize($i, i)
            } else {
                value = $i
            }
            output = output value ","
        } else if (ONE_HOT) {
            # One-hot encode categorical fields
            for (cat in categories[i]) {
                value = ($i == cat) ? 1 : 0
                output = output value ","
            }
        } else {
            output = output $i ","
        }
    }
    gsub(/,$/, "", output)
    print output
}

RS - Record Separator
---------------------
The document covers FS heavily but barely mentions RS.
RS defaults to newline, making each line one record.
You can change it to process multi-line records:

    # Process paragraph-separated records (blank line as separator)
    awk 'BEGIN { RS="" } { print NR, $1 }' file.txt

    # Use a specific delimiter as record separator
    awk 'BEGIN { RS="---" } { print NR": "$0 }' file.txt

    # In gawk, RS can be a full regex
    awk 'BEGIN { RS="[0-9]+" } { print }' file.txt


OFS - Output Field Separator
-----------------------------
The document shows OFS but misses a critical behavior:
OFS is only inserted between fields when you assign to $0
or trigger a field rebuild. Simply printing $1, $2 uses OFS,
but $0 is not rebuilt until you do an assignment.

    echo "a:b:c" | awk 'BEGIN{FS=":"; OFS=","} {print $1,$2,$3}'
    # Output: a,b,c

    echo "a:b:c" | awk 'BEGIN{FS=":"; OFS=","} {$1=$1; print $0}'
    # Output: a,b,c  (the $1=$1 trick forces field rebuild)

    echo "a:b:c" | awk 'BEGIN{FS=":"} {print $0}'
    # Output: a:b:c  (no OFS, original line printed unchanged)


FNR - Per-file record number
-----------------------------
The document uses NR throughout but FNR is important when
processing multiple files. NR counts records across all files.
FNR resets to 1 for each new file.

    # Print header from first file only, data from all files
    awk 'FNR==1 && NR!=1 { next } { print }' file1.csv file2.csv

    # The classic two-file join uses NR==FNR to detect first file
    awk 'NR==FNR { map[$1]=$2; next } { print $0, map[$1] }' keys.txt data.txt


ENVIRON - Access shell environment
-----------------------------------
    awk 'BEGIN { print ENVIRON["HOME"] }' /dev/null
    awk 'BEGIN { print ENVIRON["PATH"] }' /dev/null

    # Useful for passing config into awk without -v
    awk 'BEGIN { mode = ENVIRON["MODE"] } ...' file.txt


ARGC and ARGV - Command line arguments
---------------------------------------
    awk 'BEGIN {
        print "Number of arguments:", ARGC
        for (i=0; i<ARGC; i++) print i, ARGV[i]
    }' file1.txt file2.txt

You can modify ARGV in BEGIN to control which files get processed.


OFMT - Output format for numbers
----------------------------------
Controls how non-integer numbers are printed with print (not printf).
Default is "%.6g". Useful when you want to control float precision
without using printf everywhere.

    awk 'BEGIN { OFMT="%.2f" } { print $1 + 0.0 }' file.txt


CONVFMT - Conversion format
-----------------------------
Controls how numbers are converted to strings internally.
Default is "%.6g". Rarely needed but can cause subtle bugs
if you are comparing numbers that got converted to strings
in an intermediate step.

SECTION 4 - GETLINE PROPERLY
----------------------------------------------------------------

The document mentions getline but does not cover its four forms
or its hazards. This is tested on RHCE.

Four forms of getline:

    # Form 1: Read next line from current file into $0
    # Updates $0, NF, NR, FNR
    getline

    # Form 2: Read next line into a variable
    # Updates NR, FNR but NOT $0 or NF
    getline var

    # Form 3: Read from a file into $0
    # Updates $0, NF but NOT NR or FNR
    getline < "filename"

    # Form 4: Read from a file into a variable
    # Updates nothing except var
    getline var < "filename"

    # Form 5: Read from a command
    "command" | getline var

Always check the return value. getline returns 1 on success,
0 on end of file, -1 on error:

    while ((getline line < "file.txt") > 0) {
        print line
    }

    # Never do this - breaks on read errors:
    while (getline line < "file.txt") { print line }

getline hazard with pipes: once you open a pipe it stays open
for the rest of the script. If you call the same command in a
loop it reads sequentially, not re-running the command each time.
Close it explicitly when done:

    while (("ls /tmp" | getline fname) > 0) {
        print fname
    }
    close("ls /tmp")

SECTION 5 - AWK NUMERIC AND STRING COMPARISON GOTCHAS
----------------------------------------------------------------

Awk decides whether to compare as strings or numbers based on
context. This causes subtle bugs that the document does not warn about.

    # These look the same but behave differently:
    awk '$1 == 10'   file.txt   # numeric comparison if $1 looks like a number
    awk '$1 == "10"' file.txt   # always string comparison

    # Padding zeros: "09" == "9" is FALSE as strings, TRUE as numbers
    echo "09" | awk '{ if ($1 == 9) print "match" }'    # prints match (numeric)
    echo "09" | awk '{ if ($1 == "9") print "match" }'  # no match (string)

    # Uninitialized variables are "" and 0 simultaneously
    awk 'BEGIN { if (x == "") print "empty string" }'  # prints
    awk 'BEGIN { if (x == 0)  print "zero" }'          # also prints
    awk 'BEGIN { if (!x)      print "falsy" }'         # also prints

    # Force numeric comparison by adding 0
    awk '$1+0 > 100 { print }' file.txt

    # Force string comparison by concatenating empty string
    awk '$1"" == "100" { print }' file.txt

SECTION 8 - SED BRANCHING (more depth than the document)
----------------------------------------------------------------

The document shows the hold space but does not clearly explain
branching. Branching is how sed does conditionals and loops.

b - Branch to label (unconditional)
    sed ':top; s/aa/a/; /aa/b top' file.txt
    # Label is "top", branch back while "aa" exists (remove duplicate a's)

t - Branch to label if substitution succeeded since last t or line read
    sed ':loop; s/old/new/; t loop' file.txt
    # Keeps substituting until no more matches (for overlapping patterns)
    # t only branches if the LAST s command made a change

T - Branch to label if substitution did NOT succeed (GNU sed)
    sed 's/pattern/match/; T skip; s/match/replacement/; :skip' file.txt
    # Only runs second substitution if first one succeeded

Practical branching example - remove all blank lines including
lines with only spaces:
    sed '/^[[:space:]]*$/d' file.txt

Multi-line join with branching:
    sed ':a; N; $!ba; s/\n/ /g' file.txt
    # :a creates label "a"
    # N appends next line to pattern space
    # $!ba branches back to "a" if not last line
    # Then replaces all newlines with spaces
    # Result: entire file on one line

SECTION 9 - BRE vs ERE IN SED
----------------------------------------------------------------

The document uses -r and -E but never explains what they do.

By default sed uses BRE (Basic Regular Expressions):
    In BRE: ( ) { } + ? | must be escaped with \ to be special
    sed 's/\(foo\)\+/bar/'  file.txt   # BRE: match one or more "foo"

With -E (or -r on older GNU sed): ERE (Extended Regular Expressions):
    In ERE: ( ) { } + ? | are special WITHOUT escaping
    sed -E 's/(foo)+/bar/'  file.txt   # ERE: same thing, cleaner syntax

Practical difference:
    BRE: sed 's/[0-9]\{3\}-[0-9]\{4\}/REDACTED/'  file.txt
    ERE: sed -E 's/[0-9]{3}-[0-9]{4}/REDACTED/'   file.txt

Use -E for anything complex. BRE backslash requirements are
a common source of bugs and are hard to read.

Note: -E is POSIX standard. -r is GNU sed only (though common).
Prefer -E for portability.

SECTION 13 - LOCALE EFFECTS ON PATTERN MATCHING
----------------------------------------------------------------

The document mentions locale briefly but does not explain the
actual impact, which can bite you on production systems.

Character classes like [a-z] are locale-dependent:
    In C/POSIX locale: [a-z] matches only ASCII lowercase a through z
    In en_US.UTF-8 locale: [a-z] may match accented characters too,
    and the range boundaries depend on collation order.

This causes intermittent bugs when scripts run under different
locales on different systems.

Safe approach - use POSIX character classes instead of ranges:
    [[:alpha:]]   instead of  [a-zA-Z]
    [[:digit:]]   instead of  [0-9]
    [[:alnum:]]   instead of  [a-zA-Z0-9]
    [[:lower:]]   instead of  [a-z]
    [[:upper:]]   instead of  [A-Z]
    [[:space:]]   instead of  [ \t\n]
    [[:print:]]   any printable character

POSIX classes work consistently regardless of locale.

Force C locale for predictable behavior in scripts:
    LC_ALL=C awk '/[a-z]/' file.txt
    LC_ALL=C sed 's/[a-z]/X/g' file.txt

SECTION 14 - COMPLETE LIST OF SED HOLD SPACE COMMANDS
----------------------------------------------------------------

The document shows one hold space example but not all commands.

h   copy pattern space to hold space (overwrites hold space)
H   append pattern space to hold space (with newline separator)
g   copy hold space to pattern space (overwrites pattern space)
G   append hold space to pattern space (with newline separator)
x   exchange pattern space and hold space

Hold space starts as an empty string at the beginning of the script
and persists across lines. Pattern space is reset for each line.

Common hold space patterns:

Reverse a file (like tac):
    sed -n '1!G; $p; h' file.txt
    # G appends hold to pattern, h copies pattern to hold
    # On last line, print the accumulated result

Print the line before a match:
    sed -n '/pattern/{ x; p; x }; h' file.txt
    # h saves current line to hold each iteration
    # On match, x swaps to get previous line, print, swap back

Join every two lines:
    sed 'N; s/\n/ /' file.txt
    # N reads next line into pattern space
    # Then substitutes the embedded newline with a space

Delete trailing blank lines from a file:
    sed -e '/./{ $d; N }; P; D' file.txt

SECTION 15 - AWK ARRAY OPERATIONS (delete, in, SUBSEP)
----------------------------------------------------------------

Neither file covers these properly even though the main doc uses
delete and the `in` operator throughout without explaining them.

delete - remove array elements
-------------------------------
    # Delete a single element
    delete arr["key"]
    delete arr[1]

    # Delete the entire array (gawk and most modern awks)
    delete arr

    # POSIX-safe way to clear an entire array
    split("", arr)    # split empty string into arr wipes it

    # Check before deleting to avoid creating the element
    if ("key" in arr) delete arr["key"]

    # Practical example: remove duplicate lines preserving order
    awk '!seen[$0]++' file.txt
    # seen[$0]++ returns 0 (falsy) the first time a line is seen
    # The ! inverts it: print line only when count was 0


in - test array membership WITHOUT creating the element
---------------------------------------------------------
This is critical. Accessing arr["key"] in a condition CREATES
the element as an empty string if it does not exist. Use `in`:

    # WRONG - creates arr["key"] as "" if it does not exist
    if (arr["key"] == "") print "not found"

    # CORRECT - does not create the element
    if (!("key" in arr)) print "not found"
    if ("key" in arr) print arr["key"]

    # Loop over array keys with for...in
    for (key in arr) print key, arr[key]

    # Delete while iterating is safe in gawk
    for (key in arr) if (arr[key] == 0) delete arr[key]


SUBSEP - multi-dimensional array separator
-------------------------------------------
Awk does not have true multi-dimensional arrays. What looks like
arr[i,j] is actually arr[i SUBSEP j] where SUBSEP is a single
character (default \034, which is unlikely to appear in data).

    # These two are identical:
    arr[1,2] = "value"
    arr[1 SUBSEP 2] = "value"

    # Test membership in a multi-dimensional array
    if ((1,2) in arr) print "exists"

    # Loop and split the key back apart
    for (key in arr) {
        n = split(key, parts, SUBSEP)
        print parts[1], parts[2], arr[key]
    }

    # Practical use: frequency table by two fields
    awk '{ count[$1,$2]++ } END { for (k in count) print k, count[k] }' file.txt

SECTION 16 - AWK FLOW CONTROL MISSING FROM BOTH FILES
----------------------------------------------------------------

next - skip to the next record
--------------------------------
next immediately stops processing the current record and moves
to the next line of input. END block still runs at the end.

    # Skip comment lines and blank lines
    awk '/^#/ { next } /^$/ { next } { print }' file.txt

    # Equivalent shorthand
    awk '/^[#$]/ { next } { print }' file.txt

    # Common pattern: process first file differently
    awk 'NR==FNR { map[$1]=$2; next } { print $0, map[$1] }' a.txt b.txt
    # next here skips the second block for all lines in the first file


exit - stop processing and run END block
-----------------------------------------
exit stops reading input and jumps to the END block.
Calling exit inside END terminates immediately with no loop.

    # Stop after finding first match (like grep -m 1)
    awk '/pattern/ { print; exit }' file.txt

    # Exit with a status code
    awk '/ERROR/ { print; exit 1 }' file.txt
    echo $?   # returns 1

    # exit in END block - terminates without re-entering END
    awk 'END { print "done"; exit 0 }' file.txt

    # Practical: validate a file and bail on first error
    awk 'NF != 3 { print "Bad line:", NR; exit 1 }
         END    { if (NR > 0) print "All lines valid" }' file.txt


nextfile - skip remaining lines in current file (gawk)
-------------------------------------------------------
Jumps to the first line of the next input file.
POSIX awk does not have nextfile. gawk only.

    # Print only the first line of each file
    gawk 'FNR==1 { print FILENAME": "$0; nextfile }' *.log

    # Check if a pattern exists in any of many files
    gawk '/ERROR/ { print FILENAME; nextfile }' /var/log/*.log


BEGINFILE and ENDFILE (gawk 4.0+)
-----------------------------------
Run a block at the start and end of each input file.
Not available in POSIX awk.

    gawk 'BEGINFILE { print "=== Processing:", FILENAME }
          /ERROR/   { errors++ }
          ENDFILE   { print errors, "errors in", FILENAME; errors=0 }
         ' /var/log/*.log


----------------------------------------------------------------
SECTION 17 - AWK STRING OPERATIONS NOT EXPLAINED IN EITHER FILE
----------------------------------------------------------------

String concatenation
---------------------
Awk concatenates strings by placing them next to each other.
There is no + operator for strings. This trips up people from
other languages.

    awk 'BEGIN { a = "hello"; b = " world"; print a b }'
    # Output: hello world

    awk 'BEGIN { print "value=" 42 }'
    # Output: value=42  (42 is converted to string automatically)

    # Building a string in a loop
    awk '{ line = line " " $1 } END { print line }' file.txt

    # Concatenation vs addition - context decides
    awk 'BEGIN { a = "3"; b = "4"; print a+b, a b }'
    # Output: 7  34   (numeric context gives 7, string context gives 34)


sub() vs gsub() - the full picture
-------------------------------------
Both replace regex matches. sub() replaces only the FIRST match.
gsub() replaces ALL matches and returns the number of replacements.

    echo "aaa" | awk '{ sub(/a/, "b"); print }'    # Output: baa
    echo "aaa" | awk '{ gsub(/a/, "b"); print }'   # Output: bbb

    # & in replacement means "the matched text"
    echo "hello" | awk '{ gsub(/l/, "[&]"); print }'   # Output: he[l][l]o
    echo "hello" | awk '{ sub(/l+/, "[&]"); print }'   # Output: he[ll]o

    # To use a literal & in replacement, escape it
    echo "hello" | awk '{ gsub(/l/, "\\&"); print }'   # Output: he&&o

    # Operate on a specific field, not $0
    awk '{ gsub(/foo/, "bar", $2); print }' file.txt

    # gsub returns count of replacements
    awk '{ n = gsub(/ERROR/, "ERR"); if (n>0) print n, "replacements on line", NR }' file.txt


match() with RSTART and RLENGTH
----------------------------------
match() finds a regex in a string and sets two built-in variables:
RSTART: position of match (1-based), 0 if no match
RLENGTH: length of match, -1 if no match

    echo "foo123bar" | awk '{ if (match($0, /[0-9]+/)) print substr($0, RSTART, RLENGTH) }'
    # Output: 123

    # Check if a field contains a pattern (not just the whole line)
    awk '{ if (match($2, /^[0-9]+$/)) print "field 2 is numeric:", $2 }' file.txt

    # gawk extended match() captures groups into an array
    echo "2024-01-15" | gawk '{ match($0, /([0-9]{4})-([0-9]{2})-([0-9]{2})/, a)
                                 print "year:", a[1], "month:", a[2], "day:", a[3] }'


substr() - extract part of a string
-------------------------------------
substr(string, start [, length])
start is 1-based. If length is omitted, goes to end of string.

    awk '{ print substr($0, 1, 5) }' file.txt    # first 5 characters
    awk '{ print substr($0, 6) }' file.txt       # from position 6 to end
    awk '{ print substr($1, length($1)-1) }' file.txt  # last 2 chars of field 1

    # Combined with match() to extract a pattern
    awk '{ if (match($0,/[0-9]+/)) print substr($0,RSTART,RLENGTH) }' file.txt


index() - find substring position
------------------------------------
index(string, find) returns position of find in string, or 0.

    awk '{ if (index($0, "ERROR") > 0) print NR": "$0 }' file.txt

    # Use to check field contains a substring (faster than regex for literals)
    awk '{ if (index($2, ".conf") > 0) print }' file.txt


split() - divide a string into an array
-----------------------------------------
n = split(string, array [, separator])
Returns number of elements. array is indexed 1 to n.

    # Split a field by a delimiter
    awk '{ n = split($1, parts, ":"); for(i=1;i<=n;i++) print parts[i] }' file.txt

    # Split with a regex separator
    awk '{ n = split($0, words, /[[:space:]]+/); print n, "words" }' file.txt

    # Split an IP address
    echo "192.168.1.100" | awk '{ split($0,a,"."); print "last octet:", a[4] }'

    # split() clears the array first - safe to reuse
    awk '{ split($0, a, ":"); print a[1] }' /etc/passwd


sprintf() - format a string without printing
----------------------------------------------
Works exactly like printf but returns the formatted string
instead of printing it.

    awk '{ label = sprintf("%-10s %05d", $1, $2); print label }' file.txt

    # Build a filename from fields
    awk '{ outfile = sprintf("/tmp/output_%s.txt", $1); print $0 > outfile }' file.txt

    # Pad a number with zeros
    awk '{ printf "%s\n", sprintf("%06d", $1) }' file.txt


$NF and $(NF-1) - last fields
-------------------------------
NF is the number of fields. $NF is always the last field.
$(NF-1) is the second-to-last field.

    awk '{ print $NF }' file.txt           # print last field of each line
    awk '{ print $(NF-1) }' file.txt       # print second-to-last field
    awk 'END { print NR }' file.txt        # print total number of lines

    # Practical: get the last path component (like basename)
    echo "/usr/local/bin/awk" | awk -F/ '{ print $NF }'
    # Output: awk

    # Get process name from ps output
    ps aux | awk '{ print $NF }'


Assigning to NF - truncate or extend fields
--------------------------------------------
You can set NF to change how many fields a line has.

    # Remove last field from every line
    awk '{ NF--; print }' file.txt

    # Keep only first 3 fields
    awk '{ NF=3; print }' file.txt

    # Add a new field (field 5 when only 3 exist - pads with empty fields)
    awk '{ $5="new"; print }' file.txt


Dynamic regex vs literal regex
--------------------------------
In awk, /pattern/ is a compiled literal regex. If you store a
pattern in a variable and use it with ~, awk recompiles it each
time (slower for large files but necessary for dynamic patterns).

    # Literal regex - fast, pattern is fixed at parse time
    awk '/ERROR/ { print }' file.txt

    # Dynamic regex - pattern comes from a variable
    pattern="ERROR"
    awk -v pat="$pattern" '$0 ~ pat { print }' file.txt

    # Dynamic regex from a field
    awk 'NR==1 { pattern=$1; next } $0 ~ pattern { print }' patterns.txt data.txt

    # WRONG - this compares $0 to the string "pat", not a regex
    awk -v pat="ERROR" '$0 ~ "pat" { print }' file.txt   # matches literal "pat"

    # The ~ and !~ operators
    awk '$1 ~ /^[0-9]/ { print "starts with digit:", $1 }' file.txt
    awk '$1 !~ /^[0-9]/ { print "does not start with digit:", $1 }' file.txt


Output redirection in awk
--------------------------
The main document shows `print $1 > "output.txt"` but does not
explain the full picture.

    # Write to file (overwrites once, then appends within same awk run)
    awk '{ print > "output.txt" }' file.txt
    # Note: > does NOT truncate the file on every print. It opens once
    # and appends within the same awk invocation.

    # Append to existing file
    awk '{ print >> "output.txt" }' file.txt

    # Pipe to a command (pipe stays open for entire awk run)
    awk '{ print | "sort -u" }' file.txt

    # Write different output to different files based on content
    awk '/ERROR/ { print > "errors.txt" }
         /WARN/  { print > "warnings.txt" }
         { print > "all.txt" }' logfile.log

    # close() after writing - important if you will read the file later
    # or if you need to reopen a pipe
    awk '{ print > "tmp.txt" } END { close("tmp.txt"); while((getline line < "tmp.txt")>0) print line }' file.txt

close() - why it matters
--------------------------
    # If you pipe to sort inside a loop, close and reopen to get fresh sort
    awk '{
        print | "sort > sorted_chunk.txt"
        if (NR % 1000 == 0) {
            close("sort > sorted_chunk.txt")
        }
    }' file.txt

    # close() also flushes output files - important before reading them back
    awk '{ print $1 > "/tmp/keys.txt" }
         END { close("/tmp/keys.txt"); while ((getline k < "/tmp/keys.txt") > 0) print k }' file.txt

SECTION 18 - SED COMMANDS MISSING FROM BOTH FILES
----------------------------------------------------------------

sed n command (lowercase) - print and load next line
------------------------------------------------------
n prints the current pattern space (unless -n is active),
then reads the next line of input into the pattern space.
This is different from N which APPENDS without printing.

    # Print every other line (odd lines only)
    sed -n 'p; n' file.txt

    # Process pairs of lines: print line 1, skip line 2
    sed 'n; d' file.txt

    # Print the line AFTER a pattern match
    sed -n '/pattern/{ n; p }' file.txt

    # Substitute only on the line after a match
    sed '/pattern/{ n; s/old/new/ }' file.txt

Difference between n and N:
    n  - prints current line (if not -n), loads NEXT line into pattern space
    N  - appends next line to pattern space with embedded \n, does NOT print


sed P command (uppercase) - print first line of pattern space
--------------------------------------------------------------
When pattern space contains multiple lines (after N), P prints
only up to the first embedded newline.

    # Print only the first of each joined pair
    sed 'N; P; D' file.txt    # sliding window of 2 lines, prints each first line

    # Show context: print match and the line before it
    sed -n '/pattern/{ x; p; x; p }; h' file.txt


sed D command (uppercase) - delete first line of pattern space
---------------------------------------------------------------
Deletes up to and including the first embedded newline in pattern
space, then restarts the cycle WITHOUT reading a new line.
This is what makes sed loops work.

    # Delete blank lines at start of file
    sed '/./,$!d' file.txt

    # Remove duplicate consecutive lines (like uniq)
    sed '$!N; /^\(.*\)\n\1$/!P; D' file.txt
    # N appends next, regex checks if both lines identical, P prints if not dup, D deletes first


sed c command - replace entire matching line
---------------------------------------------
c\ replaces the entire current line (or range) with new text.
The main doc never mentions this command at all.

    # Replace a specific line entirely
    sed '5c\This is the new line 5' file.txt

    # Replace any line matching a pattern
    sed '/^ServerName/c\ServerName www.example.com' /etc/httpd/conf/httpd.conf

    # Replace a range of lines with a single line
    sed '3,7c\Lines 3 through 7 replaced' file.txt

    # Practical: update a config value regardless of current value
    sed '/^Port /c\Port 2222' /etc/ssh/sshd_config


Inserting newlines in sed replacement
---------------------------------------
Inserting a literal newline into sed output is not obvious.

    # Method 1: backslash-newline in the replacement (portable)
    sed 's/,/\
/g' file.txt

    # Method 2: use \n in replacement (GNU sed only)
    sed 's/,/\n/g' file.txt

    # Method 3: use $'\n' from bash
    sed "s/,/$'\n'/g" file.txt

    # Split CSV into one-value-per-line
    sed 's/,/\n/g' data.csv   # GNU sed

SECTION 19 - WORD BOUNDARIES IN AWK AND SED
----------------------------------------------------------------

Neither file covers word boundary matching at all.

In GNU sed:
    \b  word boundary (GNU sed extension)
    \B  non-word boundary (GNU sed extension)
    \<  start of word
    \>  end of word
    \w  word character [[:alnum:]_]  (GNU extension)
    \W  non-word character

    # Match the whole word "error" not "errors" or "noerror"
    sed -n '/\berror\b/p' file.txt
    sed -n '/\<error\>/p' file.txt   # same, older syntax

    # Replace whole word only
    sed 's/\bfoo\b/bar/g' file.txt

In gawk:
    \b  word boundary (in gawk regex)
    \B  non-word boundary
    \w  word character
    \W  non-word character
    \s  whitespace
    \S  non-whitespace

    gawk '/\<error\>/' file.txt
    gawk '/\berror\b/' file.txt

In POSIX awk (no gawk extensions):
    No \b, \w, \< or \> support. Use character classes:

    # Match "error" as a whole word using character class boundaries
    awk '/^error$|^error[^[:alnum:]_]|[^[:alnum:]_]error$|[^[:alnum:]_]error[^[:alnum:]_]/' file.txt

    # Simpler: use split or index for exact word matching
    awk '{ n=split($0,words); for(i=1;i<=n;i++) if(words[i]=="error") print }' file.txt

SECTION 20 - PROCESSING COMPRESSED FILES
----------------------------------------------------------------

Neither file mentions compressed file processing at all.
On real systems, logs and data are often gzip compressed.

With zcat / gzip:
    zcat file.gz | awk '{ print $1 }' > output.txt
    zcat file.gz | sed 's/old/new/' | gzip > modified.gz

    # Process multiple compressed files
    zcat /var/log/syslog.*.gz | awk '/ERROR/ { count++ } END { print count }'

    # Process both compressed and uncompressed together
    zcat access.log.gz access.log.1.gz | cat - access.log | awk '{ print $1 }'

With process substitution (bash):
    # Treat command output as a file argument to awk
    awk '{ print $1 }' <(zcat file.gz)

    # Join a compressed file with a normal file
    awk 'NR==FNR{map[$1]=$2;next}{print $0,map[$1]}' <(zcat keys.gz) data.txt

    # Diff two processed streams
    diff <(awk '{print $1}' file1.txt) <(awk '{print $1}' file2.txt)

With bzip2 / xz:
    bzcat file.bz2 | awk '{ print $1 }'
    xzcat file.xz  | sed 's/old/new/'

SECTION 21 - WHAT AWK AND SED CANNOT DO
----------------------------------------------------------------

Neither file covers the limits of these tools. Knowing when
something is simply not possible saves you hours of fighting.

Awk and sed cannot do:
    - Lookahead: (?=...) - not supported in either tool
    - Lookbehind: (?<=...) - not supported in either tool
    - Named capture groups: (?P<name>...) - not supported
    - Non-greedy matching: *? or +? - not supported
    - Recursive patterns - not supported
    - True in-memory sort without piping to sort
    - Network I/O (gawk has limited support via /inet/ but it is not portable)
    - Binary file processing reliably (text tools, not binary tools)
    - True random access within a file (forward-only streaming)

For these, use:
    perl -ne 'print if /(?<=prefix)pattern/'  file.txt   # lookahead/behind
    grep -P 'pattern'                                     # PCRE in grep
    python3, ruby for complex regex and data structures

The awk regex engine is based on POSIX ERE (Extended Regular
Expressions). The sed regex engine uses POSIX BRE by default
or ERE with -E. Neither uses PCRE.

SECTION 22 - FILENAME VARIABLE AND MULTI-FILE PATTERNS
----------------------------------------------------------------

FILENAME - the current input file being processed
--------------------------------------------------
    # Print filename before each group of matches
    awk '/ERROR/ { print FILENAME": "$0 }' /var/log/*.log

    # Process each file separately - print header per file
    awk 'FNR==1 { print "\n=== File:", FILENAME, "===" } { print }' *.txt

    # Count matches per file
    awk '/pattern/ { count[FILENAME]++ }
         END { for (f in count) print count[f], f }' *.log

    # In BEGIN, FILENAME is undefined (no file being read yet)
    # In END, FILENAME is the name of the LAST file processed

NR vs FNR for multi-file logic:
    NR  = total records read across ALL files (never resets)
    FNR = records read in the CURRENT file (resets for each file)

    # Add line numbers restarting at 1 for each file
    awk '{ print FNR, $0 }' file1.txt file2.txt

    # Add sequential line numbers across all files
    awk '{ print NR, $0 }' file1.txt file2.txt

    # Detect when a new file starts
    awk 'FNR==1 && NR!=1 { print "---" } { print }' file1.txt file2.txt
